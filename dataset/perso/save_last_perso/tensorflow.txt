'''
Here is an example of how you might use Python to load data into Google Cloud Platform (GCP):
'''

#Copy code
# Import the necessary libraries
import os
from google.cloud import storage

# Set the environment variable for the GCP project
os.environ["GOOGLE_CLOUD_PROJECT"] = "my-gcp-project"

# Create a client for the GCP Storage API
storage_client = storage.Client()

# Define the bucket where the data should be stored
bucket = storage_client.bucket("my-gcp-bucket")

# Define the file where the data is stored
file = "./data.csv"

# Upload the file to the bucket
blob = bucket.blob("data.csv")
blob.upload_from_filename(file)
